### **IV**.原理层

```rust
/**

 * Definition for singly-linked list.

 * class ListNode {

 *     int val;

 *     ListNode next;

 *     ListNode(int x) {

 *         val = x;

 *         next = null;

 *     }

 * }

 */

public class Solution {

    public ListNode detectCycle(ListNode head) {

         if (head == null) {

            return null;

        }

        ListNode slow = head, fast = head;

        while (fast != null) {

            slow = slow.next;

            // fast 能遇到null，证明没有环gg

            if (fast.next!=null){

                fast = fast.next.next;

            }else{

                return null;

            }

            if (fast==slow){

                ListNode ptr = head;

                while(ptr!=slow){

                    ptr = ptr.next;

                    slow=slow.next;

                }

                return ptr;

            }

        }

        return null;

    }

}

```
