## 3. 展开说明  
### 链表定义
```Java
public class ListNode {
    // 结点的值
    int val;

    // 下一个结点
    ListNode next;

    // 节点的构造函数(无参)
    public ListNode() {
    }

    // 节点的构造函数(有一个参数)
    public ListNode(int val) {
        this.val = val;
    }

    // 节点的构造函数(有两个参数)
    public ListNode(int val, ListNode next) {
        this.val = val;
        this.next = next;
    }
}
```
#### rust
```rust
#[derive(PartialEq, Eq, Debug)]

pub struct ListNode {

    pub val: i32,

    pub next: Option<Box<ListNode>>,

}

  

impl ListNode {

    #[inline]

    pub fn new(val: i32) -> Self {

        ListNode { next: None, val }

    }

}
```
