```shell

# 创建自定义 bridge 网络

docker network create \

    --driver bridge \

    --subnet=172.20.0.0/16 \

    --gateway=172.20.0.1 \

    my-bridge-network

  

# 启动两个容器，连接到自定义网络

docker run -d \

    --name custom-container1 \

    --network my-bridge-network \

    my-nginx

  

docker run -d \

    --name custom-container2 \

    --network my-bridge-network \

    my-nginx

  

# 现在可以通过容器名称访问

docker exec -it custom-container1 curl http://custom-container2

```
