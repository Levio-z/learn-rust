```shell

# 使用 host 网络运行 Nginx

docker run -d \

    --name nginx-host \

    --network host \

    my-nginx

  

# 直接通过主机的 80 端口访问

curl http://localhost:80

  

# 因为使用了 host 网络，容器直接使用主机的 80 端口, 所以当我们再次启动一个 Nginx 容器时，会报端口冲突的错误

docker run -d \

    --name nginx-host-2 \

    --network host \

    my-nginx

docker logs nginx-host-2

```

### 文件
```
FROM nginx:alpine

  

# 安装 curl

RUN apk update && apk add --no-cache curl

  

# 保留原始镜像的默认配置

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]
```
