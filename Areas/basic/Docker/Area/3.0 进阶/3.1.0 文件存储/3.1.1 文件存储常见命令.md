### Volume
```
docker volume ls # 列出所有卷
docker volume create nginx_data # 创建一个 Docker volume
docker run -d --name web-volume -p 8081:80 -v nginx_data:/usr/share/nginx/html nginx # 运行 Nginx 容器并挂载卷
docker volume inspect nginx_data # 查看卷的详细信息
docker volume rm nginx_data # 清理卷
docker volume rm $(docker volume ls -q) #删除所有卷
```
