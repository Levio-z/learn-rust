### 基础模板
>改编自：https://github.com/tikv/tikv/blob/master/rustfmt.toml
```
# =========================================
# 语言版本与实验特性
# =========================================
style_edition = "2024"          # 使用 Rust 2024 Edition 风格进行格式化
unstable_features = true        # 启用 Rustfmt 不稳定功能（需要 nightly）

# =========================================
# 注释与文档规范
# =========================================
comment_width = 80              # 注释最大宽度 80 字符
wrap_comments = true            # 自动换行普通注释
format_code_in_doc_comments = true  # 格式化文档注释中的代码块
normalize_comments = true       # 统一注释缩进和多余空格
normalize_doc_attributes = true # 统一文档注解排版（如 #[doc = "..."]）

# =========================================
# 宏相关
# =========================================
format_macro_bodies = true      # 格式化宏定义体内部代码
format_macro_matchers = true    # 格式化宏匹配模式（matcher）

# =========================================
# 导入与通配符
# =========================================
condense_wildcard_suffixes = true      # 压缩导入语句中的多余通配符 `::*`
imports_granularity = "Crate"          # 导入按 crate 级别分组，如 use my_crate::{mod1, mod2};
group_imports = "StdExternalCrate"     # 导入分组顺序：标准库 → 外部 crate → 本地 crate

# =========================================
# 换行与语法简写
# =========================================
newline_style = "Unix"           # 使用 Unix 风格换行符 \n
use_field_init_shorthand = true  # 结构体初始化使用简写：X { a, b }
use_try_shorthand = true         # 使用 ? 简写错误处理：do_something()? 
```
