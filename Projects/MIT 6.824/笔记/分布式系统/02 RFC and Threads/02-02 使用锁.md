```go
package main

import (
	"fmt"
	"math/rand"
	"sync"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano()) // 初始化随机数种子

	count := 0        // 记录投票成功的数量
	finished := 0     // 记录完成投票的goroutine数量
	var mu sync.Mutex // 创建一个互斥锁

	// 创建10个goroutine来模拟投票
	for i := 0; i < 10; i++ {
		go func() {
			vote := requestVote() // 请求投票
			mu.Lock()            // 进入临界区，加锁
			defer mu.Unlock()     // 函数执行完毕后自动解锁
			if vote {
				count++ // 如果投票成功，计数器加1
			}
			finished++ // 无论投票成功与否，完成投票的goroutine数量加1
		}()
	}

	// 等待直到收到5个以上的投票或所有goroutine都完成投票
	for {
		mu.Lock()
		if count >= 5 || finished == 10 {
			break
		}
		mu.Unlock()
	}

	// 检查是否收到5个以上的投票
	if count >= 5 {
		fmt.Println("received 5+ votes!")
	} else {
		fmt.Println("lost")
	}
}

// requestVote 模拟投票请求，这里假设它随机返回true或false
func requestVote() bool {
	// 随机等待一段时间来模拟网络延迟或投票处理时间
	time.Sleep(time.Duration(rand.Intn(100)) * time.Millisecond)
	return rand.Intn(2) == 1 // 随机返回true或false来模拟投票结果
}
```
- 不使用锁，count可能读取的不是最新值
- go会在使用锁之后使用内存屏障

缺点
- 处理器自旋，什么都不做
	- 可以增加sleep等待，但是坏处是这里该休眠多久并不好把控。
