import (
	"fmt"

	"time"
)

	count := 0        // 记录投票成功的数量
	finished := 0     // 记录完成投票的goroutine数量

	// 创建10个goroutine来模拟投票
	for i := 0; i < 10; i++ {
		go func() {
			vote := requestVote() // 请求投票，这里假设返回true或false
			if vote {
				count++ // 如果投票成功，计数器加1
			}
			finished++ // 无论投票成功与否，完成投票的goroutine数量加1
		}()
	}

	// 等待直到收到5个以上的投票或所有goroutine都完成投票
	for count < 5 && finished != 10 {
		// 这里只是空循环等待，没有sleep，可能会导致 busy-waiting
	}

	// 检查是否收到5个以上的投票
	if count >= 5 {
		fmt.Println("received 5+ votes!")
	} else {
		fmt.Println("lost")
	}
}
