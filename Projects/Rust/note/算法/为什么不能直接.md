```rust
impl Solution {

    pub fn remove_elements(head: Option<Box<ListNode>>, val: i32) -> Option<Box<ListNode>> {

        let mut dummy = Box::new(ListNode{next:None,val:0});

        let mut cur = &mut dummy;
		
        while let Some(ref mut nxt) = cur.next{
			// ----------- `cur.next` is borrowed here
            if nxt.val == val{
				// cannot assign to `cur.next` because it is borrowed
				// `cur.next` is assigned to here but it was already borrowed | borrow later used here
                cur.next = nxt.next.take();

            }else{

                cur = nxt;

                //

            }

        }

        dummy.next

    }

}
```
- cur本身上有一个借用
- nxt是cur.next上的一个借用
