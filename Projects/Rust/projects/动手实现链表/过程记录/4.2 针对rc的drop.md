impl<T> Drop for List<T> {
    fn drop(&mut self) {
        let mut head = self.head.take(); // 把 self.head 取出，置为 None
        while let Some(node) = head {
            if let Ok(mut node) = Rc::try_unwrap(node) {
                head = node.next.take(); // 继续下一个节点
            } else {
                break; // 有其他引用，停止
            }
        }
    }
}
